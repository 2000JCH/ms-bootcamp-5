// 추상 클래스 (공통 설계도 역할)
public abstract class Shape {
    protected String color;

    // 생성자
    public Shape(String color) {
        this.color = color;
    }

    // 추상 메소드 (구현은 자식 클래스에서)
    public abstract double getArea();
    public abstract double getPerimeter();

    // 일반 메소드 (공통 기능 제공)
    public void displayColor() {
        System.out.println("색상: " + color);
    }
}

// 구체 클래스 (원)
public class Circle extends Shape {
    private double radius;

    public Circle(String color, double radius) {
        super(color);
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return Math.PI * radius * radius;
    }

    @Override
    public double getPerimeter() {
        return 2 * Math.PI * radius;
    }
}

// 실행 클래스
public class AbstractExam {
    public static void main(String[] args) {
        Shape circle = new Circle("빨강", 5.0);

        circle.displayColor();                 // → "색상: 빨강"
        System.out.println("넓이: " + circle.getArea());         // → 78.5398...
        System.out.println("둘레: " + circle.getPerimeter());   // → 31.4159...
    }
}
```

// 추상 클래스 Animal
public abstract class Animal {
    private String name;

    // 일반 메소드
    public void eat() {
        System.out.println("먹는다.");
    }

    public void walk() {
        System.out.println("걷는다.");
    }

    // 추상 메소드 (각 동물이 다르게 구현해야 함)
    public abstract void makeSound();
}

// Dog 클래스 (Animal 상속)
public class Dog extends Animal {
    @Override
    public void makeSound() {
        System.out.println("멍멍!");
    }
}

// Cat 클래스 (Animal 상속)
public class Cat extends Animal {
    @Override
    public void makeSound() {
        System.out.println("야옹~");
    }
}

// 실행 클래스
public class AbstractAnimalTest {
    public static void main(String[] args) {
        Animal dog = new Dog();
        dog.eat();          // → "먹는다."
        dog.walk();         // → "걷는다."
        dog.makeSound();    // → "멍멍!"

        Animal cat = new Cat();
        cat.eat();          // → "먹는다."
        cat.walk();         // → "걷는다."
        cat.makeSound();    // → "야옹~"
    }
}

package day07;

// 수상 인터페이스
public interface SmartDevice {
    void turnOn();
    void turnOff();

    // default 메소드 (Java 8+)
    default void reset() {
        System.out.println("기본 설정으로 초기화합니다.");
        turnOff();
        turnOn();
    }

    // static 메소드 (Java 8+)
    static void showInfo() {
        System.out.println("스마트 기기 인터페이스 v1.0");
    }
}

// 구현 클래스
public class SmartPhone implements SmartDevice {
    @Override
    public void turnOn() {
        System.out.println("스마트폰 전원 ON");
    }

    @Override
    public void turnOff() {
        System.out.println("스마트폰 전원 OFF");
    }

    // default 메소드 오버라이딩 가능
    @Override
    public void reset() {
        System.out.println("스마트폰 초기화 중...");
    }
}

// 실행 클래스
public class InterfaceTest {
    public static void main(String[] args) {
        SmartDevice device = new SmartPhone();
        device.turnOn();   // "스마트폰 전원 ON"
        device.turnOff();  // "스마트폰 전원 OFF"
        device.reset();    // "스마트폰 초기화 중..."

        SmartDevice.showInfo(); // static 메소드 호출
    }
}
```
